CREATE OR REPLACE TRIGGER ODS_CLIENTTYPE_BIR_TRI BEFORE INSERT OR UPDATE ON ODS_CLIENTTYPE
FOR EACH ROW
DECLARE
	   tmpVar NUMBER;
/******************************************************************************
   COMPANY: ORACLE
   PROJECT: MIS
   NAME:	ODS_CLIENTTYPE_BIR_TRI
   PURPOSE: Default values for several columns of the table
   NOTES:

   REVISIONS:
   Ver        Date        Author           Description
   ---------  ----------  ---------------  ------------------------------------
   1.0        4/01/2007   GPAPOUTS         1. Created this trigger.
******************************************************************************/
BEGIN
	 IF INSERTING THEN
	 	IF :NEW.CLIENTTYPE_SK IS NULL THEN
          SELECT ODS_CLIENTTYPE_SEQ.NEXTVAL INTO tmpVar FROM dual;
          :NEW.CLIENTTYPE_SK := tmpVar;
		END IF;
		:NEW.CLIENTTYPE_CREATIONDATE	:= SYSDATE;
     	:NEW.CLIENTTYPE_CREATIONUSER	:= nvl(v('APP_USER'), user);
--
		INSERT INTO ODS_CLIENT_TARG (CLIENT_DISCRIMINATOR, CLIENT_DISCRDESCR, CLIENT_CODE, CLIENT_DESCR)
			   VALUES (1, 'Τύπος Πελάτη', :NEW.CLIENTTYPE_CODE, :NEW.CLIENTTYPE_DESCR);
	 ELSIF UPDATING THEN
	 	  IF :NEW.CLIENTTYPE_CODE != :OLD.CLIENTTYPE_CODE THEN
		  	 BEGIN
			 	  INSERT INTO ODS_CLIENT_TARG (CLIENT_DISCRIMINATOR, CLIENT_DISCRDESCR, CLIENT_CODE, CLIENT_DESCR)
			   	  		 VALUES (1, 'Τύπος Πελάτη', :NEW.CLIENTTYPE_CODE, :NEW.CLIENTTYPE_DESCR);
			 EXCEPTION
			 	 WHEN DUP_VAL_ON_INDEX THEN
				 	  NULL;
			 END;
	 	  END IF;
	 	  IF :NEW.CLIENTTYPE_DESCR != :OLD.CLIENTTYPE_DESCR AND :NEW.CLIENTTYPE_CODE = :OLD.CLIENTTYPE_CODE THEN
	 	  	 UPDATE ODS_CLIENT_TARG SET CLIENT_DESCR = :NEW.CLIENTTYPE_DESCR
		  		 WHERE CLIENT_DISCRIMINATOR = 1
				 AND CLIENT_CODE = :NEW.CLIENTTYPE_CODE;
		  END IF;
	 END IF;
EXCEPTION
		 WHEN OTHERS THEN
-- Consider logging the error and then re-raise
   			RAISE;
END  ODS_CLIENTTYPE_BIR_TRI;
/
SHOW ERRORS;
