CREATE OR REPLACE TRIGGER ODS_NETRANGE_BIR_TRI BEFORE INSERT ON ODS_NETRANGE
FOR EACH ROW
DECLARE
	   tmpVar NUMBER;
/******************************************************************************
   COMPANY: ORACLE
   PROJECT: MIS
   NAME:	ODS_NETRANGE_BIR_TRI
   PURPOSE: Default values for several columns of the table
   NOTES:

   REVISIONS:
   Ver        Date        Author           Description
   ---------  ----------  ---------------  ------------------------------------
   1.0        8/01/2007   GPAPOUTS         1. Created this trigger.
******************************************************************************/
BEGIN
	 IF INSERTING THEN
	 	IF :NEW.NETRANGE_SK IS NULL THEN
          SELECT ODS_NETRANGE_SEQ.NEXTVAL INTO tmpVar FROM dual;
          :NEW.NETRANGE_SK := tmpVar;
		END IF;
		:NEW.NETRANGE_CREATIONDATE	:= SYSDATE;
     	:NEW.NETRANGE_CREATIONUSER := nvl(v('APP_USER'), user);
	 END IF;
EXCEPTION
		 WHEN OTHERS THEN
-- Consider logging the error and then re-raise
   			RAISE;
END ODS_NETRANGE_BIR_TRI;
/
SHOW ERRORS;
