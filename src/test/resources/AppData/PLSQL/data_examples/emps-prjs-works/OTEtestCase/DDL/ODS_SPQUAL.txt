CREATE OR REPLACE TRIGGER ODS_SPQUAL_BIUR_TRI BEFORE INSERT OR UPDATE ON ODS_SPQUAL
FOR EACH ROW
DECLARE
	   tmpVar NUMBER;
/******************************************************************************
   COMPANY: ORACLE
   PROJECT: MIS
   NAME:	ODS_SPQUAL_BIUR_TRI
   PURPOSE: Default values for several columns of the table
   NOTES:

   REVISIONS:
   Ver        Date        Author           Description
   ---------  ----------  ---------------  ------------------------------------
   1.0        4/01/2007   GPAPOUTS         1. Created this trigger.
******************************************************************************/
BEGIN
	 IF INSERTING THEN
	 	IF :NEW.SPQUAL_SK IS NULL THEN
          SELECT ODS_SPQUAL_SEQ.NEXTVAL INTO tmpVar FROM dual;
          :NEW.SPQUAL_SK := tmpVar;
		END IF;
		:NEW.SPQUAL_CREATIONDATE	:= SYSDATE;
     	:NEW.SPQUAL_CREATIONUSER	:= nvl(v('APP_USER'), user);
--
		INSERT INTO ODS_SPQUAL_TARG (SPQUAL_DISCRIMINATOR, SPQUAL_DISCRDESCR, SPQUAL_CODE, SPQUAL_DESCR)
			   VALUES (2, 'A?aio Oa?yocoao/?ieuocoao', :NEW.SPQUAL_CODE, :NEW.SPQUAL_DESCR);
	 ELSIF UPDATING THEN
	 	IF :NEW.SPQUAL_CODE != :OLD.SPQUAL_CODE THEN
		   BEGIN
 		   		INSERT INTO ODS_SPQUAL_TARG (SPQUAL_DISCRIMINATOR, SPQUAL_DISCRDESCR, SPQUAL_CODE, SPQUAL_DESCR)
			   		   VALUES (2, 'A?aio Oa?yocoao/?ieuocoao', :NEW.SPQUAL_CODE, :NEW.SPQUAL_DESCR);
		   EXCEPTION
		   		WHEN DUP_VAL_ON_INDEX THEN
					 NULL;
		   END;
		END IF;
	 	IF :NEW.SPQUAL_DESCR = :OLD.SPQUAL_DESCR AND :NEW.SPQUAL_CODE = :OLD.SPQUAL_CODE THEN
	 	   UPDATE ODS_SPQUAL_TARG SET SPQUAL_DESCR = :NEW.SPQUAL_DESCR
		  		 WHERE SPQUAL_DISCRIMINATOR = 2
				 AND SPQUAL_CODE = :NEW.SPQUAL_CODE;
		END IF;
	 END IF;
EXCEPTION
		 WHEN OTHERS THEN
-- Consider logging the error and then re-raise
   			RAISE;
END  ODS_SPQUAL_BIUR_TRI;
/
SHOW ERRORS;
