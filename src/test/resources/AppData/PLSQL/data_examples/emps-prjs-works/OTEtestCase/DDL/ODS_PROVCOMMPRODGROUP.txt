CREATE OR REPLACE TRIGGER ODS_PROVCOMMPRODGROUP_BIR_TRI BEFORE INSERT OR UPDATE ON ODS_PROVCOMMPRODGROUP
FOR EACH ROW
DECLARE
	   tmpVar NUMBER;
/******************************************************************************
   COMPANY: ORACLE
   PROJECT: MIS
   NAME:	ODS_PROVCOMMPRODGROUP_BIR_TRI
   PURPOSE: Default values for several columns of the table
   NOTES:

   REVISIONS:
   Ver        Date        Author           Description
   ---------  ----------  ---------------  ------------------------------------
   1.0        4/01/2007   GPAPOUTS         1. Created this trigger.
******************************************************************************/
BEGIN
	 IF INSERTING THEN
	 	IF :NEW.PROVCOMMPRODGROUP_SK IS NULL THEN
          SELECT ODS_PROVCOMMPRODGROUP_SEQ.NEXTVAL INTO tmpVar FROM dual;
          :NEW.PROVCOMMPRODGROUP_SK := tmpVar;
		END IF;
		:NEW.PROVCOMMPRODGROUP_CREATIONDATE	:= SYSDATE;
     	:NEW.PROVCOMMPRODGROUP_CREATIONUSER	:= nvl(v('APP_USER'), user);
--
		INSERT INTO ODS_PROVCOMMPROD_TARG (PROVCOMMPROD_DISCRIMINATOR, PROVCOMMPROD_DISCRDESCR, PROVCOMMPROD_CODE, PROVCOMMPROD_DESCR)
			   VALUES (1, 'IiUaao Ai?inee?i ?niuuioui', :NEW.PROVCOMMPRODGROUP_CODE, :NEW.PROVCOMMPRODGROUP_DESCR);
	 ELSIF UPDATING THEN
	 	  IF :NEW.PROVCOMMPRODGROUP_CODE != :OLD.PROVCOMMPRODGROUP_CODE THEN
		  	 BEGIN
			 	  INSERT INTO ODS_PROVCOMMPROD_TARG (PROVCOMMPROD_DISCRIMINATOR, PROVCOMMPROD_DISCRDESCR, PROVCOMMPROD_CODE, PROVCOMMPROD_DESCR)
			   	  		 VALUES (1, 'IiUaao Ai?inee?i ?niuuioui', :NEW.PROVCOMMPRODGROUP_CODE, :NEW.PROVCOMMPRODGROUP_DESCR);
			 EXCEPTION
			 	 WHEN DUP_VAL_ON_INDEX THEN
				 	  NULL;
			 END;
		  END IF;
	 	  IF :NEW.PROVCOMMPRODGROUP_DESCR != :OLD.PROVCOMMPRODGROUP_DESCR AND :NEW.PROVCOMMPRODGROUP_CODE = :OLD.PROVCOMMPRODGROUP_CODE THEN
	 	  	 UPDATE ODS_PROVCOMMPROD_TARG SET PROVCOMMPROD_DESCR = :NEW.PROVCOMMPRODGROUP_DESCR
		  		 WHERE PROVCOMMPROD_DISCRIMINATOR = 1
				 AND PROVCOMMPROD_CODE = :NEW.PROVCOMMPRODGROUP_CODE;
		  END IF;
	 END IF;
EXCEPTION
		 WHEN OTHERS THEN
-- Consider logging the error and then re-raise
   			RAISE;
END  ODS_PROVCOMMPRODGROUP_BIR_TRI;
/
SHOW ERRORS;
