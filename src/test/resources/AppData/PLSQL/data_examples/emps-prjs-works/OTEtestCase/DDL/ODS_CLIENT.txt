CREATE OR REPLACE TRIGGER ODS_CLIENT_BIR_TRI BEFORE INSERT OR UPDATE ON ODS_CLIENT
FOR EACH ROW
DECLARE
	   tmpVar NUMBER;
/******************************************************************************
   COMPANY: ORACLE
   PROJECT: MIS
   NAME:	ODS_CLIENT_BIR_TRI
   PURPOSE: Default values for several columns of the table
   NOTES:

   REVISIONS:
   Ver        Date        Author           Description
   ---------  ----------  ---------------  ------------------------------------
   1.0        4/01/2007   GPAPOUTS         1. Created this trigger.
******************************************************************************/
BEGIN
	 IF INSERTING THEN
	 	IF :NEW.CLIENT_SK IS NULL THEN
          SELECT ODS_CLIENT_SEQ.NEXTVAL INTO tmpVar FROM dual;
          :NEW.CLIENT_SK := tmpVar;
		END IF;
		:NEW.CLIENT_CREATIONDATE	:= SYSDATE;
     	:NEW.CLIENT_CREATIONUSER	:= nvl(v('APP_USER'), user);
--
		INSERT INTO ODS_CLIENT_TARG (CLIENT_DISCRIMINATOR, CLIENT_DISCRDESCR, CLIENT_CODE, CLIENT_DESCR)
			   VALUES (2, 'Πελάτης', :NEW.CLIENT_CODE, :NEW.CLIENT_DESCR);
	 ELSIF UPDATING THEN
	 	  IF :NEW.CLIENT_CODE != :OLD.CLIENT_CODE THEN
		  	 BEGIN
			 	  INSERT INTO ODS_CLIENT_TARG (CLIENT_DISCRIMINATOR, CLIENT_DISCRDESCR, CLIENT_CODE, CLIENT_DESCR)
			   	  		 VALUES (2, 'Πελάτης', :NEW.CLIENT_CODE, :NEW.CLIENT_DESCR);
			 EXCEPTION
			 	 WHEN DUP_VAL_ON_INDEX THEN
				 	  NULL;
			 END;
	 	  END IF;
	 	  IF :NEW.CLIENT_DESCR != :OLD.CLIENT_DESCR AND :NEW.CLIENT_CODE = :OLD.CLIENT_CODE THEN
	 	  	 UPDATE ODS_CLIENT_TARG SET CLIENT_DESCR = :NEW.CLIENT_DESCR
		  		 WHERE CLIENT_DISCRIMINATOR = 2
				 AND CLIENT_CODE = :NEW.CLIENT_CODE;
		  END IF;
	 END IF;
EXCEPTION
		 WHEN OTHERS THEN
-- Consider logging the error and then re-raise
   			RAISE;
END  ODS_CLIENT_BIR_TRI;
/
SHOW ERRORS;
