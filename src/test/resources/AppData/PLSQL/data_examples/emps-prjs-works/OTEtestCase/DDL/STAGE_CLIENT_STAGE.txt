CREATE OR REPLACE TRIGGER STAGE_CLIENT_STAGE_BIR_TRI BEFORE INSERT ON STAGE_CLIENT_STAGE
FOR EACH ROW
DECLARE
	   tmpVar NUMBER;
/******************************************************************************
   COMPANY: ORACLE
   PROJECT: MIS
   NAME:	STAGE_CLIENT_BIR_TRI
   PURPOSE: Default values for several columns of the table
   NOTES:

   REVISIONS:
   Ver        Date        Author           Description
   ---------  ----------  ---------------  ------------------------------------
   1.0        15/1/2007   GPAPOUTS         1. Created this trigger.
******************************************************************************/
BEGIN
	 IF :NEW.CLIENT_SK IS NULL THEN
           SELECT STAGE_CLIENT_STAGE_SEQ.NEXTVAL INTO tmpVar FROM dual;
          :NEW.CLIENT_SK := tmpVar;
     END IF;
EXCEPTION
		 WHEN OTHERS THEN
-- Consider logging the error and then re-raise
   			RAISE;
END  STAGE_CLIENT_STAGE_BIR_TRI;
/
SHOW ERRORS;
